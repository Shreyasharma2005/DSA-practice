class Solution {
    public int[] nextSmallerElements(int[] arr) {
        int[] result=new int[arr.length];
        Stack<Integer> stack=new Stack<>();
        Array.fill(result,-1);
        for(int i=0;i<arr.length;i++){
            while(!stack.isEmpty() && arr[i] < stack.peek()) {
                int index = stack.pop();
                result[index] = arr[i];
            }
            stack.push(i);
        }
        return result;
    }
}
