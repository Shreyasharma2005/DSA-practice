class ArrayQueue {
    int[] arr;
    int front;
    int rear;

    public ArrayQueue(int size) {
        arr = new int[size];
        front = 0;
        rear = -1;
    }

    public void push(int x) {
        if (rear == arr.length - 1) {
            System.out.println("Queue overflow");
            return;
        }
        rear++;
        arr[rear] = x;
    }

    public int pop() {
        if (front > rear) {
            System.out.println("Queue underflow");
            return -1;
        }
        int value = arr[front];
        front++;
        return value;
    }

    public int peek() {
        if (front > rear) {
            System.out.println("Queue is empty");
            return -1;
        }
        return arr[front];
    }

    public boolean isEmpty() {
        return front > rear;
    }
}
